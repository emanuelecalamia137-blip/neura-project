import openai
import tkinter as tk
import pyttsx3

# === INSERISCI LA TUA API KEY QUI ===
openai.api_key = "INSERISCI_LA_TUA_API_KEY"

# === Funzione per chiedere a ChatGPT ===
def chiedi_a_chatgpt():
    domanda = domanda_entry.get()
    if not domanda:
        risposta_label.config(text="Scrivi una domanda!")
        return

    risposta_label.config(text="Sto pensando...")

    try:
        response = openai.ChatCompletion.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": domanda}]
        )
        risposta = response.choices[0].message.content.strip()
        risposta_label.config(text=risposta)
        parla(risposta)
    except Exception as e:
        risposta_label.config(text=f"Errore: {e}")

# === Funzione per leggere la risposta a voce ===
def parla(testo):
    engine = pyttsx3.init()
    engine.say(testo)
    engine.runAndWait()

# === GUI con tkinter ===
finestra = tk.Tk()
finestra.title("Intervista con ChatGPT")
finestra.geometry("600x400")

tk.Label(finestra, text="Fai una domanda a ChatGPT:", font=("Arial", 14)).pack(pady=10)
domanda_entry = tk.Entry(finestra, font=("Arial", 12), width=60)
domanda_entry.pack(pady=5)

tk.Button(finestra, text="Invia", command=chiedi_a_chatgpt, font=("Arial", 12)).pack(pady=10)

risposta_label = tk.Label(finestra, text="", wraplength=550, font=("Arial", 12), justify="left")
risposta_label.pack(pady=10)

finestra.mainloop()
