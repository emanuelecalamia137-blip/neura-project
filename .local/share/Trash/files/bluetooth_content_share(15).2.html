nano ~/chatgpt.sh

#!/bin/bash

# === INSERISCI LA TUA API KEY QUI ===
API_KEY="INSERISCI_LA_TUA_API_KEY"

# === INPUT DOMANDA ===
read -p "Fai la tua domanda a ChatGPT: " domanda

# === CHIAMATA API ===
risposta=$(curl -s https://api.openai.com/v1/chat/completions \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $API_KEY" \
  -d '{
    "model": "gpt-3.5-turbo",
    "messages": [{"role": "user", "content": "'"$domanda"'"}]
  }' | jq -r '.choices[0].message.content')

# === STAMPA RISPOSTA ===
echo -e "\nChatGPT risponde:\n$risposta"

# === VOCE (richiede espeak installato) ===
espeak "$risposta"
